# Incompatibilidade de tipos 3 com string(nome) #

@DDL
Table aluno {
	$aluno->aincrement('id');
	$aluno->string('nome');
	$aluno->string('cpf');
	$aluno->integer('idade')->unsigned();
	$aluno->genTimestamps();
}

@endDDL

@DML
$aluno = [
	3,
	"123.456.789-0",
	22,
	"mat.santos.freitas@gmail.com"
];
@endDML
