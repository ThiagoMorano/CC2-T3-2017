@DDL
Table autor {
	$autor->aincrement('id');
	$autor->integer('livro_id');
	$autor->string('nome');
	$autor->string('email');
	$autor->integer('idade')->unsigned();
	$autor->string('cpf');

	Relationships{
		$autor->hasMany('livro');
	}
}

Table livro {
	$livro->aincrement('id');
	$livro->integer('autor_id');
	$livro->string('nome');
	$livro->string('editora');
	$livro->string('descricao');
	$livro->date('ano_lancamento');
	$livro->integer('edicao')->unsigned();

	Relationships{
		$livro->hasMany('autor');
	}
}
@endDDL
