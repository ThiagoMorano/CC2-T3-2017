CREATE TABLE aluno (
	id INT AUTOINCREMENT PRIMARY KEY NOT NULL,
	universidade_id INT NOT NULL, nome VARCHAR(255) NOT NULL,
	cpf VARCHAR(255) NOT NULL, idade INT UNSIGNED NOT NULL,
	created_at TIMESTAMP DEFAULT '0000-00-00 00:00:00' NOT NULL,
	updated_at TIMESTAMP DEFAULT '0000-00-00 00:00:00' NOT NULL,
	FOREIGN KEY (universidade_id) REFERENCES universidade(id)
);

CREATE TABLE universidade (
	id INT AUTOINCREMENT PRIMARY KEY NOT NULL,
	ano_fundacao INT UNSIGNED NOT NULL,
	created_at TIMESTAMP NOT NULL DEFAULT '0000-00-00 00:00:00',
	updated_at TIMESTAMP NOT NULL DEFAULT '0000-00-00 00:00:00'
);
